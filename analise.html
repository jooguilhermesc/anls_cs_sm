Para poder trabalhar com este conjunto de dados realizei algumas operaes de limpeza de dados, como remover valores nulos e os smbolos R$ e % dos dados e trabalha-los como nmeros decimais, por fim agrupei os anos que possuam mais de um valor de salrio pela mdia de salrio daquele ano.

url = 'http://www.fetapergs.org.br/index.php/2015-07-27-16-46-22/tabelas-salario-minimo' #'https://previdenciarista.com/blog/tabela-historica-dos-salarios-minimos/'
sr_hist_sm = pd.read_html(url, header=[0])
df_sm = sr_hist_sm[0]
df_sm = df_sm[['Vigncia','Salrio Mnimo','Reajuste Salrio Mnimo']]
df_sm['Ano'] = df_sm.Vigncia.apply(lambda x: x[-4:])
df_sm = df_sm.drop(columns=['Vigncia'])
df_sm = df_sm.rename(columns={'Salrio Mnimo':'vl_sm','Reajuste Salrio Mnimo':'vl_reaj_sm'})
df_sm = df_sm.dropna(subset=['vl_reaj_sm'])
df_sm['vl_sm'] = df_sm['vl_sm'].apply(lambda x: x.replace('R$ ','').replace('.','').replace(',','.'))
df_sm['vl_reaj_sm'] = df_sm['vl_reaj_sm'].apply(lambda x: x.replace('%','').replace(',','.'))
df_sm['vl_sm'] = df_sm['vl_sm'].astype(np.float64)
df_sm['vl_reaj_sm'] = df_sm['vl_reaj_sm'].astype(np.float64)
df_sm = df_sm.groupby('Ano').mean()
